import { Meta, IconGallery, IconItem } from "@storybook/blocks";
import * as Icons from "./";
import { createRenderer } from "fela";
import { RendererProvider } from "react-fela";
import expandShorthand from "fela-plugin-expand-shorthand";
import sortClassNames from "fela-sort-classnames";
import sortMediaQueryMobileFirst from "fela-sort-media-query-mobile-first";
import prefixer from "fela-plugin-prefixer";

export const renderer = createRenderer({
  plugins: [prefixer(), expandShorthand(true)],
  enhancers: [sortClassNames(), sortMediaQueryMobileFirst()],
});

export const MdxWrapper = ({ children, renderer }) => {
  return <RendererProvider renderer={renderer}>{children}</RendererProvider>;
};

# Icons

These icons are used in the bootstrap documentation. They are included in the library.
This is not an exhaustive list of icons, it is strongly advised to
consider [Bootstrap Icons](https://icons.getbootstrap.com/).

The `CheckIcon` is rendered here with a fill and stroke color of `black`, the default icon is transparent with a stroke color of `#fff`.
The `SearchIcon` is off center and aligned to the left.
